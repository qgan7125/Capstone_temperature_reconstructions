Load data in

There are three data sets with low, intermediate, and high errors.

data_low <- read_csv("../Data/Dataset_S1_Mar8.csv")
data_intermediate <- read_csv("../Data/Dataset_S2_Mar8.csv")
data_high <- read_csv("../Data/Dataset_S3_Mar8.csv")

Bayesian linear regression

Using the Bayesian linear regression without error The model setup is is stan/Bayesian_linear_model.stan

Data with low error

stan_data_low <- list(N = nrow(data_low), Temperature = data_low$Temperature, D47 = data_low$D47)

rstan_options(auto_write = TRUE)
model_low <- stan(file = "stan/Bayesian_linear_model.stan", data = stan_data_low, iter = 1000, chains = 4, seed = 1234)

results_low <- rstan::extract(model_low)
cat("Median Intercept", median(results_low$alpha))

Comparing result to the true parameters

Plot confident interval area of the median slope/intercept of bootstrapping results And comparing it to the true slope/intercept

I use the true data points to plot.

data_low %>% 
  ggplot(aes(x = x_TRUE, y=y_TRUE)) + 
  geom_ribbon(aes(ymin = summary(model_low)$summary[2,4] * x_TRUE + summary(model_low)$summary[1,8] , ymax = summary(model_low)$summary[2,8] * x_TRUE + summary(model_low)$summary[1,4]), fill = "orange") +
  geom_point() +
  geom_abline(slope = 0.0369, intercept = 0.268, color = 'red') +
  geom_abline(slope = median(results_low$beta), intercept = median(results_low$alpha), color = 'yellow') + 
  labs(title = "Bayesian model with low error data") + 
  xlab("True Temperature") + 
  ylab("True D47")

Bayesian linear regression Results summary

medianIntercept <- c(median(results_low$alpha), median(results_intermediate$alpha), median(results_high$alpha))
se_mean_Intercept <- c(summary(model_low)$summary[1,2], summary(model_intermediate)$summary[1,2], summary(model_high)$summary[1,2])
medianSlope <- c(median(results_low$beta), median(results_intermediate$beta), median(results_high$beta))
se_mean_Slope <- c(summary(model_low)$summary[2,2], summary(model_intermediate)$summary[2,2], summary(model_high)$summary[2,2])

mat <- array(c(medianIntercept, se_mean_Intercept, medianSlope, se_mean_Slope), dim = c(3, 4))
colnames(mat) <- c("median_Intercept", "se_mean_Intercept", "median_Slope", "se_mean_slope")
rownames(mat) <- c("Low Error", "Intermediate Error", "High Error")

mat %>% 
  kable()
